/*SetCurrencyValues function to list out the requried coins to reach the provided currency*/
  setCurrencyValues(amount:any) {
    var coinVal:any = [];
    let calc:any;
    for(let i = 0; i< this.coins.length; i++){
      calc = Math.floor(amount / this.coins[i]);
